procedure TTreeNTNode.SetOverlayIndex(Value: Integer);

var Item: TTVItem;

begin
  FOverlayIndex := Value;
  with Item do
  begin
    Mask := TVIF_STATE or TVIF_HANDLE;
    stateMask := TVIS_OVERLAYMask;
    hItem := ItemId;
    state := IndexToOverlayMask(Value + 1);
  end;
  TreeView_SetItem(Handle, Item);
End;
