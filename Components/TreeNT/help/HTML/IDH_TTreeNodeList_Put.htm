procedure TTreeNodeList.Put(Index: Integer; Item: TTreeNTNode);

begin
  if (Index < 0) or (Index >= FCount) then
  {$ifdef DFS_COMPILER_3_UP}
    Error(SListIndexError, Index);
  {$else}
    Error(LoadStr(SListIndexError), Index);
  {$endif}
  FList[Index] := Item;
End;
