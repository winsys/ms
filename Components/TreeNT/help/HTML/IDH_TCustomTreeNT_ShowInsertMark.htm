Implementation

procedure TCustomTreeNT.ShowInsertMark(Node: TTreeNTNode; After: Boolean);

begin
  if assigned(Node) then TreeView_SetInsertmark(Handle, Integer(Node.ItemID), After)
                    else TreeView_SetInsertmark(Handle, 0, After);
End;
