function TTreeNTNode.GetPrev: TTreeNTNode;

var Node : TTreeNTNode;

begin
  Result := GetPrevSibling;
  if assigned(Result) then
  begin
    Node := Result;
    repeat
      Result := Node;
      Node := Result.GetLastChild;
    until Node = nil;
  end
  else
    if FParent <> FOwner.FRoot then Result := FParent;
End;
