function TTreeNTNodes.GetNode(ItemID: HTreeItem): TTreeNTNode;

var Item : TTVItem;

begin
  Result := nil;
  if ItemID = nil then Exit; // special case: FRoot

  with Item do
  begin
    hItem := ItemId;
    Mask := TVIF_PARAM;
  end;
  if TreeView_GetItem(Handle, Item) then Result := TTreeNTNode(Item.lParam);
End;
