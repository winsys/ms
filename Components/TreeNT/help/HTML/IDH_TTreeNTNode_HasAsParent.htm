function TTreeNTNode.HasAsParent(Value: TTreeNTNode): Boolean;

begin
  if assigned(Value) then
  begin
    if Self = Value then Result := True
                    else
      if (FParent <> FOwner.FRoot) then Result := FParent.HasAsParent(Value)
                                   else Result := False;
  end
  else Result := True;
End;
