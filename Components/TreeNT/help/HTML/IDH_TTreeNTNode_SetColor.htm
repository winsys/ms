Implementation

procedure TTreeNTNode.SetColor(AValue: TColor);

begin
  if FColor <> AValue then
  begin
    FColor := AValue;
    FParentColor := False;
    if not IsUpdating then FOwner.Repaint(Self);
  end;
End;
