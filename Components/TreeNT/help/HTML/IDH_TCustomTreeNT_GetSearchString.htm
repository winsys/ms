function TCustomTreeNT.GetSearchString: String;

var Count  : Integer;

begin
  Count := Integer(TreeView_GetISearchString(Handle, nil));
  if Count > 0 then
  begin
    SetString(Result, nil, Count);
    TreeView_GetISearchString(Handle, PChar(Result));
  end
  else Result := '';
End;
