function TCustomTreeNT.GetNodeFromItem(const Item: TTVItem): TTreeNTNode;

begin
  with Item do
    if (State and TVIF_PARAM) <> 0 then Result := Pointer(lParam)
                                   else Result := Items.GetNode(hItem);
End;
