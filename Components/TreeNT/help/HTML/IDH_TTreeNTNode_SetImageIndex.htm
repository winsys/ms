procedure TTreeNTNode.SetImageIndex(Value: Integer);

var Item: TTVItem;

begin
  FImageIndex := Value;
  with Item do
  begin
    Mask := TVIF_IMAGE or TVIF_HANDLE;
    hItem := ItemId;
    iImage := I_IMAGECALLBACK;
  end;
  TreeView_SetItem(Handle, Item);
End;
