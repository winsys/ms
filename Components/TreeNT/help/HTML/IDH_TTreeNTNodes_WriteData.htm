procedure TTreeNTNodes.WriteData(Stream: TStream);

var I        : Integer;
    Node     : TTreeNTNode;
    NodeInfo : TNodeInfo;

begin
  Stream.WriteBuffer(FRootCheckType, SizeOf(FRootCheckType));
  I := FRoot.Count;
  Stream.WriteBuffer(I, SizeOf(I));
  Node := GetFirstNode;
  while Node <> nil do
  begin
    Node.WriteData(Stream, @NodeInfo);
    Node := Node.GetNextSibling;
  end;
End;
