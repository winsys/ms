procedure TCustomTreeNT.Edit(const Item: TTVItem);

var S    : String;
    Node : TTreeNTNode;

begin
  with Item do
    if pszText <> nil then
    begin
      S := pszText;
      Node := GetNodeFromItem(Item);
      if assigned(FOnEdited) then FOnEdited(Self, Node, S);
      if assigned(Node) then
      begin
        Node.Text := S;
        if assigned(Designer) then Designer.Modified;
      end;  
    end;
End;
