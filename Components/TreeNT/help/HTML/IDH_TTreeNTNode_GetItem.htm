function TTreeNTNode.GetItem(Index: Integer): TTreeNTNode;

begin
  if Index > (FChildList.Count - 1) then
  begin
    Result := nil;
    TreeNTError(SListIndexError);
  end
  else Result := FChildList[Index];
End;
