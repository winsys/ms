Applies to
TCustomTreeNT

Declaration
Procedure SetScrollTime(Value: Integer);

Implementation

procedure TCustomTreeNT.SetScrollTime(Value: Integer);

begin
  if Value <> FScrollTime then
  begin
    FScrollTime := Value;
    if FScrollTime < 100 then FScrollTime := 100;
    Treeview_SetScrollTime(Handle, FScrollTime);
  end;
End;
