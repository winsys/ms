function TTreeNTNode.GetIntegralHeight: Integer;

var Item: PTVItemEx;

begin
  Result := 0;
  New(Item);
  try
    if assigned(ItemID) then  // special case FRoot of TTreeNTNodes
    begin
      Item.Mask := TVIF_INTEGRAL;
      Item.hItem := ItemId;
      if TreeView_GetItem(Handle, PTVItem(Item)^) then Result := Item.iIntegral;
    end;
  finally
    Dispose(Item);
  end;
End;
