function TTreeNTNode.GetNextVisible: TTreeNTNode;

var I, Index : Integer;

begin
  Result := nil;
  with FOwner do
  begin
    FillCache;
    Index := FItemCache.IndexOf(Self) + 1;
    for I := Index to FItemCache.Count - 1 do
      if FItemCache[I].IsVisible then
      begin
        Result := FItemCache[I];
        Break;
      end;
  end;
End;
