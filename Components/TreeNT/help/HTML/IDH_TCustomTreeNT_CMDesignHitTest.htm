procedure TCustomTreeNT.CMDesignHitTest(var Message: TCMDesignHitTest);

var HitInfo : THitTests;                              

begin
  HitInfo := GetHitTestInfoAt(Message.XPos, Message.YPos);
  if (htOnItem in HitInfo) or (htOnButton in HitInfo) then Message.Result := 1
                                                      else
  begin
    Message.Result := 0;
    if IsEditing then Selected.EndEdit(True);
  end;
End;
