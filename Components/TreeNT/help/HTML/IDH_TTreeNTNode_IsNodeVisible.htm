function TTreeNTNode.IsNodeVisible: Boolean;

var AParent : TTreeNTNode;

begin
  AParent := FParent;
  while assigned(AParent) and AParent.FExpanded do AParent := AParent.FParent;
  if assigned(AParent) then Result := False
                       else Result := True;
End;
